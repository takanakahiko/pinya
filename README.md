```bash

$ export CONSUMER_SECRET=YYYYYYYYYYYYYYYYYY
$ export ACCESS_TOKEN=ZZZZZZZZZZZZZZZZZZZZZZZZ
$ export ACCESS_TOKEN_SECRET=HOGEEEEEEEEEEEEEEEEEEEEEE

$ npm i
$ npm run start
# Check the ID from the tweet list
[
  {
    id_str: 'XXXXXXXXXXXXXXXXXXXXXX',
    full_text: 'tweet text tweet text tweet text'
  },
  {
    id_str: 'YYYYYYYYYYYYYYYYYYYYYY',
    full_text: 'tweet text2 tweet text2 tweet text2'
  },
  ...
]

$ npm run start XXXXXXXXXXXXXXXXXXXXXX
# Tweet will be pinned to your profile
```
